// Java: userSe(USER_SE) 기반 분기
LoginVO user = (LoginVO) EgovUserDetailsHelper.getAuthenticatedUser();
switch (user.getUserSe()) {
    case "USR": /* 업무사용자: 관리/운영 */ break;
    case "GNR": /* 일반회원: 서비스 */     break;
    case "ENT": /* 기업회원: 기업서비스 */  break;
}
